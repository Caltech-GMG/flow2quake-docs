 15def _true_h(H, h):
 16    """
 17    Define the true physical value for the thickness of the gas
 18    (if h<0 : h=0, if h>H : h=H, else : keep value of h)
 19    H must be provided
 20    """
 21    return (H + (h + abs(h)) / 2 - abs(H - (h + abs(h)) / 2)) / 2

 24def _theoretical_injected_mass(t: float, flow_sources: list[FlowSource]) -> float:

 26    Calculating theoritical injected (or extracted) mass. Positive for injected, negative for extracted
 27
 28    :param float t: time in seconds
 29    :param list[FlowSource] flow_sources: flow sources
 30    :param float sigma: standard deviation of flow rate
 31    """
 32    amount = 0
 33    for source in flow_sources:
 34        # NOTE: This was unused for some reason
 35        # New_inj_profile = []
 36        if len(source.flow_history) > 0:
 37            for flow_rate, time_start, time_stop in source.flow_history:
 38                if time_stop <= t:
 39                    amount += flow_rate * (time_stop - time_start)
 40                elif time_start <= t:
 41                    amount += flow_rate * (t - time_start)

153    """
154    Determine if we inject or not at time t for a given injection profile (which corresponds to a source)
155
156    :param t: time
157    :param Inj_profile: injection profile (start, end)

166def _s(flow_sources: list[FlowSource], t: float, ME1: FunctionSpace, sigma: float):
167    """
168    Define source function at time t from the list with sources data (Sources_g)
169    ME1, sigma must be provided
170
171    :param list[FlowSource] Sources_g: list of sources
172    :param float t: time
173    :param dolfin.FunctionSpace ME1: function space
174    :param float sigma: standard deviation of flow rate
175    """
176    # alpha = 1/sigma**2 ; K = 1/(np.pi*sigma**2)
177    f_g = "0"
178    # s_g_ = Function(ME1)
179    for source in flow_sources:
180        x = source.x
181        y = source.y
182        for flow_rate, time_begin, time_end in source.flow_history:
183            bin = _bin_inject(t, [time_begin, time_end])
184            f_g += (
185                "+"
186                + str(bin)
187                + "*"
188                + str(flow_rate)
189                + "*K*exp(-alpha*((x[0]-"
190                + str(x)
191                + ")*(x[0]-"
192                + str(x)
193                + ")+(x[1]-"
194                + str(y)
195                + ")*(x[1]-"
196                + str(y)
197                + ")))"
198            )
199    s_g = Expression(f_g, K=1 / (np.pi * sigma**2), alpha=1 / sigma**2, degree=2)
200    return s_g

203def _create_mesh(
204    domain: Polygon,
205    Nm: int,
206    Nhr: int,
207    Nst: int,
208    flow_sources: list[FlowSource],
209    sigma: float,
210    phi: float,
211    sg: float,
212    rho_g0: float,
213    H0: float,
214) -> Mesh:
215    """
216    Create mesh of the problem for FEM simulation
217
218    :param mshr.Polygon domain: domain of the problem
219    :param int Nm: number of initial mesh refinements
220    :param int Nhr: number of mesh refinements for hydraulic radius
221    :param int Nst: number of mesh refinements for sources
222    :param Sources_g: gas sources
223    :param sigma: sigma
224    :param float phi: porosity
225    :param float sg: gas saturation
226    :param float rho_g0: gas density
227    :param float H0: thickness of the reservoir
228    :returns dolfin.Mesh: mesh
229    """
230    problem_mesh = generate_mesh(domain, Nm)
231    for _ in range(Nhr):
232        cell_markers = MeshFunction("bool", problem_mesh, problem_mesh.topology().dim())
233        for c in cells(problem_mesh):
234            cell_markers[c] = False
235            for source in flow_sources:
236                p = source.point
237                Injected_mass = np.matmul(
238                    np.array([e[0] for e in source.flow_history]),
239                    np.array([[e[2] - e[1]] for e in source.flow_history]),
240                )
241                Hydraulic_radius = np.sqrt(Injected_mass / (phi * sg * rho_g0 * H0))
242                if c.midpoint().distance(p) < 4 * Hydraulic_radius:
243                    cell_markers[c] = True
244        problem_mesh = refine(problem_mesh, cell_markers)
245    for _ in range(Nst):
246        cell_markers = MeshFunction("bool", problem_mesh, problem_mesh.topology().dim())
247        for c in cells(problem_mesh):
248            cell_markers[c] = False
249            for source in flow_sources:
250                p = source.point
251                if c.midpoint().distance(p) < 5 * sigma:
252                    cell_markers[c] = True
253        problem_mesh = refine(problem_mesh, cell_markers)
254    return problem_mesh

333class VFEModel(NonlinearProblem):
334    def __init__(self, a, L, bcs):
335        """
336        :param a: bilinear form
337        :param L: linear form
338        :param bcs: boundary conditions
339        """
340        self.L = L
341        self.a = a
342        self.bcs = bcs
343        NonlinearProblem.__init__(self)
344
345    def F(self, b, x):
346        """
347        Computes the residual vector
348        :param b: vector
349        :param x: vector
350        """
351        assemble(self.L, tensor=b)
352        for bc in self.bcs:
353            bc.apply(b, x)
354
355    def J(self, A, x):
356        """
357        Compute the Jacobian matrix
358        :param A: matrix
359        :param x: vector
360        """
361        assemble(self.a, tensor=A)
362        for bc in self.bcs:
363            bc.apply(A)

366    def from_problem(
367        water: Liquid,
368        gas: Gas,
369        dt: float,
370        t: float,
371        flow_sources: list[FlowSource],
372        ME1: FunctionSpace,
373        H,
374        phi,
375        cr,
376        u,
377        p,
378        h,
379        p0,
380        h0,
381        du,
382        bcs,
383        sigma,
384        v_p,
385        v_h,
386        k,
387        g=9.81,
388    ):
389        """
390        Define the variational problem for a multi-phase flow model using the finite element method.
391
392        :param base.Liquid water: water object
393        :param base.Gas gas: gas object
394        :param float dt: time step
395        :param float t: time
396        :param list[FlowSource] Sources_g: gas sources
397        :param dolfin.FunctionSpace ME1: function space
398        :param dolfin.Function H: thickness function
399        :param dolfin.Function phi: porosity function
400        :param float cr: compressibility ratio
401        :param dolfin.Function u: solution function
402        :param dolfin.Function p: pressure function
403        :param dolfin.Function h: thickness function
404        :param float p0: initial pressure (Pa)
405        :param float h0: initial thickness
406        :param dolfin.Function du: solution increment
407        :param list bcs: boundary conditions
408        :param float sigma: standard deviation of flow rate
409        :param dolfin.Function v_p: test function for pressure
410        :param dolfin.Function v_h: test function for thickness
411        """
412        s_g_ = Function(ME1)  # Source term
413        s_g_.interpolate(_s(flow_sources, t, ME1, sigma))
414
415        L_p = (
416            (
417                phi
418                * (
419                    (cr * gas.rho(p) + gas.c * gas.rho_0) * gas.s * h * (p - p0)
420                    + gas.rho(p) * gas.s * (h - _true_h(H, h0))
421                )
422                * v_p
423                * dx
424            )
425            + (
426                dt
427                * gas.rho(p)
428                * k
429                * gas.kr
430                / gas.mu(p)
431                * h
432                * inner(nabla_grad(p) - gas.rho_0 * g * nabla_grad(h), nabla_grad(v_p))
433                * dx
434            )
435            - (dt * s_g_ * v_p * dx)
436        )
437
438        L_h = (
439            phi
440            * (
441                (cr * water.rho(p) + water.rho_0 * water.c) * (H - gas.s * h) * (p - p0)
442                - water.rho(p) * gas.s * (h - _true_h(H, h0))
443            )
444            * v_h
445            * dx
446        ) + (
447            dt
448            * water.rho(p)
449            * k
450            * water.kr
451            / water.mu(p)
452            * (H - h)
453            * inner(nabla_grad(p) - water.rho_0 * g * nabla_grad(h), nabla_grad(v_h))
454            * dx
455        )
456
457        L = L_h + L_p
458        # Compute directional derivative about u in the direction of du (Jacobian)
459        J = derivative(L, u, du)
460        return VFEModel(J, L, bcs)
