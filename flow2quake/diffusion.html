541            self.solver.parameters["convergence_criterion"] = "incremental"
542            self.solver.parameters["relative_tolerance"] = 1e-8
543            self.solver.parameters["absolute_tolerance"] = 1e-7
544            self.solver.parameters["maximum_iterations"] = 20
545        else:
546            self.solver = solver
547
548        # create problem mesh
549        self.problem_mesh = _create_mesh(
550            domain=self.domain.polygon,
551            Nm=self._params.num_initial_mesh_refinements,
552            Nhr=self._params.num_hydraulic_radius_refinements,
553            Nst=self._params.num_source_refinements,

556            phi=np.mean(self.reservoir.porosity_field.flatten()),

559            H0=np.min(self.reservoir.thickness_field.flatten()),

564        mixed_element = MixedElement([P1, P1])
565        self.ME = FunctionSpace(self.problem_mesh, mixed_element)
566        self.ME1 = FunctionSpace(self.problem_mesh, P1)

568        # Attribute functions

610
611        self.current_mass_found = (

617

619        self.pressure_grid_init = griddata(

621            self.reservoir.pressure_field,

623            method="linear",  # Linear interpolation

628            self.reservoir.pressure_field,
629            self.reservoir.thickness_points,
630            self.reservoir.thickness_field,
631            self.ME1,
632        )

634        self._outline_points = []
635        for i in range(len(self.domain.points) - 1):
636            point1 = self.domain.points[i]
637            point2 = self.domain.points[i + 1]
638            X = np.linspace(point1[0], point2[0], self._params.boundary_points)
639            Y = np.linspace(point1[1], point2[1], self._params.boundary_points)
640            for j in range(len(X)):
641                self._outline_points.append([X[j], Y[j]])

644            tolerance = self._params.boundary_conditions_tolerance
645            if on_boundary:
646                for vertex in self._outline_points:
647                    if near(x[0], vertex[0], tolerance) and near(
648                        x[1], vertex[1], tolerance

650                        return True
651            return False
